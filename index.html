异常处理

程序错误的捕获和处理

Python异常处理示例


# 1. 基本的try-except结构
try:
    num = int("abc")          # 尝试将字符串转换为数字
except ValueError:            # 捕获类型转换错误
    print("无法将字符串转换为数字")

# 2. 处理除零错误
def divide(a, b):
    try:
        result = a / b        # 尝试除法运算
        print(f"结果是: {result}")
    except ZeroDivisionError:
        print("错误：除数不能为零")

divide(10, 2)                 # 正常运算
divide(10, 0)                 # 触发除零错误

# 3. 多个异常的处理
def get_user_info(data):
    try:
        name = data["name"]   # 尝试访问字典
        age = int(data["age"])
        print(f"{name}今年{age}岁")
    except KeyError:
        print("错误：缺少必要的信息")
    except ValueError:
        print("错误：年龄必须是数字")

# 测试不同情况
get_user_info({"name": "小明", "age": "16"})     # 正常情况
get_user_info({"name": "小华"})                  # 缺少age
get_user_info({"name": "小红", "age": "abc"})    # 年龄格式错误

# 4. 使用finally子句
def open_file(filename):
    try:
        file = open(filename, "r", encoding="utf-8")
        content = file.read()
        return content
    except FileNotFoundError:
        print("错误：文件不存在")
    finally:
        print("操作完成")      # 无论是否出错都会执行

# 测试文件操作
open_file("test.txt")         # 尝试打开不存在的文件

文件操作

文件的读写和基本处理

Python文件操作示例


# 1. 写入文本文件
def write_notes():
    try:
        with open("notes.txt", "w", encoding="utf-8") as file:
            file.write("今天天气晴朗\n")          # 写入一行
            file.write("适合出去散步")           # 写入另一行
        print("笔记保存成功")
    except Exception as e:
        print(f"写入失败：{e}")

# 2. 读取文本文件
def read_notes():
    try:
        with open("notes.txt", "r", encoding="utf-8") as file:
            content = file.read()                # 读取全部内容
            print("文件内容：")
            print(content)
    except FileNotFoundError:
        print("文件不存在")

# 3. 逐行读取
def read_lines():
    try:
        with open("notes.txt", "r", encoding="utf-8") as file:
            print("逐行读取：")
            for line in file:                    # 逐行处理
                print(f"-> {line.strip()}")      # 去除换行符
    except FileNotFoundError:
        print("文件不存在")

# 4. 追加内容
def append_note(new_note):
    try:
        with open("notes.txt", "a", encoding="utf-8") as file:
            file.write(f"\n{new_note}")         # 追加新内容
        print("追加成功")
    except Exception as e:
        print(f"追加失败：{e}")

# 测试文件操作
write_notes()                                   # 创建并写入文件
read_notes()                                    # 读取整个文件
read_lines()                                    # 逐行读取
append_note("晚上去看电影")                      # 追加内容
read_notes()                                    # 再次读取

面向对象

类和对象的基本使用

Python面向对象示例


# 1. 定义简单的类
class Student:
    def __init__(self, name, age):      # 初始化方法
        self.name = name                # 设置姓名
        self.age = age                  # 设置年龄
    
    def introduce(self):                # 类的方法
        print(f"大家好，我是{self.name}，今年{self.age}岁")

# 创建学生对象
student1 = Student("小明", 12)           # 创建对象
student1.introduce()                    # 调用方法

# 2. 带属性的类
class Rectangle:
    def __init__(self, width, height):
        self.width = width              # 宽度属性
        self.height = height            # 高度属性
    
    def calculate_area(self):           # 计算面积
        return self.width * self.height
    
    def show_info(self):                # 显示信息
        area = self.calculate_area()
        print(f"矩形面积：{area}平方米")

# 使用矩形类
rect = Rectangle(5, 3)                  # 创建矩形对象
rect.show_info()                       # 显示面积

# 3. 继承示例
class Animal:
    def __init__(self, name):
        self.name = name
    
    def speak(self):
        pass                            # 基类方法

class Dog(Animal):                      # 继承Animal类
    def speak(self):                    # 重写speak方法
        return f"{self.name}说：汪汪!"

class Cat(Animal):                      # 继承Animal类
    def speak(self):                    # 重写speak方法
        return f"{self.name}说：喵喵!"

# 使用继承类
dog = Dog("小黑")
cat = Cat("咪咪")
print(dog.speak())                      # 调用狗的speak方法
print(cat.speak())                      # 调用猫的speak方法
